-- Step 1: Create Tables

CREATE TABLE users (
    user_id INTEGER PRIMARY KEY,
    name TEXT,
    signup_date DATE
);

CREATE TABLE orders (
    order_id INTEGER PRIMARY KEY,
    user_id INTEGER,
    amount INTEGER
);

CREATE TABLE activity (
    user_id INTEGER,
    login_count INTEGER
);

-- 1. INNER JOIN users and orders
SELECT users.name, orders.amount
FROM users
INNER JOIN orders
ON users.user_id = orders.user_id;


-- 2. Total order amount per user
SELECT users.name, SUM(orders.amount) AS total_order_amount
FROM users
INNER JOIN orders
ON users.user_id = orders.user_id
GROUP BY users.name;


-- 3. LEFT JOIN users with activity
SELECT users.name, activity.login_count
FROM users
LEFT JOIN activity
ON users.user_id = activity.user_id;


-- 4. Users who never placed an order
SELECT users.name
FROM users
LEFT JOIN orders
ON users.user_id = orders.user_id
WHERE orders.order_id IS NULL;


-- 5. Combine all three tables
SELECT 
users.user_id,
users.name,
SUM(orders.amount) AS total_order_amount,
activity.login_count
FROM users
LEFT JOIN orders
ON users.user_id = orders.user_id
LEFT JOIN activity
ON users.user_id = activity.user_id
GROUP BY users.user_id, users.name, activity.login_count;


-- 6. Users who placed orders greater than 400
SELECT users.name, orders.amount
FROM users
INNER JOIN orders
ON users.user_id = orders.user_id
WHERE orders.amount > 400;


-- 7. Top 3 users by total order amount
SELECT users.name, SUM(orders.amount) AS total_order_amount
FROM users
INNER JOIN orders
ON users.user_id = orders.user_id
GROUP BY users.name
ORDER BY total_order_amount DESC
LIMIT 3;


-- 8. Sort by total order amount and login count
SELECT 
users.name,
SUM(orders.amount) AS total_order_amount,
activity.login_count
FROM users
LEFT JOIN orders
ON users.user_id = orders.user_id
LEFT JOIN activity
ON users.user_id = activity.user_id
GROUP BY users.name, activity.login_count
ORDER BY total_order_amount DESC, activity.login_count DESC;


-- 9. Average order amount per user
SELECT users.name, AVG(orders.amount) AS avg_order_amount
FROM users
INNER JOIN orders
ON users.user_id = orders.user_id
GROUP BY users.name;


-- 10. Combined query
SELECT 
users.user_id,
users.name,
SUM(orders.amount) AS total_order_amount,
activity.login_count
FROM users
LEFT JOIN orders
ON users.user_id = orders.user_id
LEFT JOIN activity
ON users.user_id = activity.user_id
GROUP BY users.user_id, users.name, activity.login_count
ORDER BY total_order_amount DESC
LIMIT 3;
