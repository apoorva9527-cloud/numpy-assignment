-- Step 1: Create Table

CREATE TABLE students (
    id INTEGER PRIMARY KEY,
    name TEXT,
    score INTEGER,
    passed BOOLEAN,
    class INTEGER
);

-- 1. Retrieve all columns
SELECT * FROM students;


-- 2. Retrieve name and score
SELECT name, score FROM students;


-- 3. Students who scored above 80
SELECT * FROM students
WHERE score > 80;


-- 4. Students who scored above 80 AND passed
SELECT * FROM students
WHERE score > 80 AND passed = TRUE;


-- 5. Students who scored above 85 OR belong to class 9
SELECT * FROM students
WHERE score > 85 OR class = 9;


-- 6. Students who did NOT pass
SELECT * FROM students
WHERE NOT passed;


-- 7. Distinct classes
SELECT DISTINCT class
FROM students;


-- 8. Top 5 highest scoring students
SELECT * FROM students
ORDER BY score DESC
LIMIT 5;


-- 9. Sort by class ASC and score DESC
SELECT * FROM students
ORDER BY class ASC, score DESC;


-- 10. Alias for score
SELECT name, score AS final_score
FROM students;


-- 11. Combined query
SELECT name, score
FROM students
WHERE score > 75
ORDER BY score DESC
LIMIT 3;
